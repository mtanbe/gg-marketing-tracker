Connect to Google Sheets

All projects sync live via Google Sheets so your whole team sees the same data in real time. Paste your Apps Script Web App URL below to connect.

Make sure you are opening this file directly in your browser (double-click the .html file), not inside Claude.ai. The Google Sheets connection only works when the file is opened locally.
1

Go to sheets.google.com → create a sheet named "GG Marketing Tracker" → share with your team (Edit access).

2

Open Extensions → Apps Script → delete existing code → paste the script below → Save.

3

Deploy → New deployment → Web App → Execute as: Me → Who has access: Anyone → Deploy → copy the URL.

4

Paste the URL below and click Connect. Share this .html file with your team — they paste the same URL and are instantly synced.

Apps Script — paste into Extensions → Apps Script 
const SHEET_NAME = "Projects";
const HEADERS = ["id","account","am","name","type","stage","gd","cw","start","deadline","notes","createdAt","onlineDate"];

function getOrCreateSheet(ss) {
  let sheet = ss.getSheetByName(SHEET_NAME);
  if (!sheet) {
    sheet = ss.insertSheet(SHEET_NAME);
    sheet.appendRow(HEADERS);
  }
  return sheet;
}

function respond(callback, obj) {
  const json = JSON.stringify(obj);
  return ContentService
    .createTextOutput(callback ? callback + "(" + json + ")" : json)
    .setMimeType(callback ? ContentService.MimeType.JAVASCRIPT : ContentService.MimeType.JSON);
}

function doGet(e) {
  const cb = e.parameter.callback || "";
  const action = e.parameter.action || "read";
  const ss = SpreadsheetApp.getActiveSpreadsheet();
  const sheet = getOrCreateSheet(ss);

  if (action === "add") {
    const p = JSON.parse(e.parameter.p);
    sheet.appendRow(HEADERS.map(h => p[h] !== undefined ? p[h] : ""));
    return respond(cb, { ok: true });
  }

  if (action === "update") {
    const p = JSON.parse(e.parameter.p);
    const data = sheet.getDataRange().getValues();
    for (let i = 1; i < data.length; i++) {
      if (String(data[i][0]) === String(p.id)) {
        sheet.getRange(i+1, 1, 1, HEADERS.length)
          .setValues([HEADERS.map(h => p[h] !== undefined ? p[h] : "")]);
        break;
      }
    }
    return respond(cb, { ok: true });
  }

  if (action === "delete") {
    const id = e.parameter.id;
    const data = sheet.getDataRange().getValues();
    for (let i = 1; i < data.length; i++) {
      if (String(data[i][0]) === String(id)) {
        sheet.deleteRow(i + 1);
        break;
      }
    }
    return respond(cb, { ok: true });
  }

  if (action === "draft") {
    try {
      const to = e.parameter.to;
      const cc = e.parameter.cc || "";
      const subject = e.parameter.subject;
      const body = e.parameter.body;
      const ccList = cc ? cc.split(",").map(s => s.trim()).filter(Boolean) : [];
      GmailApp.createDraft(to, subject, body, { cc: ccList.join(",") });
      return respond(cb, { ok: true });
    } catch(err) {
      return respond(cb, { ok: false, error: err.toString() });
    }
  }

  if (action === "push") {
    const props = PropertiesService.getScriptProperties();
    props.setProperty("push_msg", e.parameter.msg || "");
    props.setProperty("push_ts", String(Date.now()));
    return respond(cb, { ok: true });
  }

  if (action === "getpush") {
    const props = PropertiesService.getScriptProperties();
    const msg = props.getProperty("push_msg") || "";
    const ts = props.getProperty("push_ts") || "0";
    return respond(cb, { ok: true, msg: msg, ts: ts });
  }

  // Default: read all rows
  const data = sheet.getDataRange().getValues();
  const headers = data[0];
  const rows = data.slice(1).map(r => {
    const obj = {};
    headers.forEach((h, i) => obj[h] = String(r[i]));
    return obj;
  });
  return respond(cb, { ok: true, rows: rows });
}

function doPost(e) {
  return doGet(e);
}
